102 static std::vector<std::string> split(const std::string& s, const std::string& toRemove = "None",
103 const std::string& toReplace = "None")
104 {
105 std::string t = s;
106
107 // first remove or replace
108 if(toRemove != "None")
109 {
110 if(toReplace == "None")
111 {
112 for(int i=0; i<toRemove.length(); i++)
113 t.erase(std::remove(t.begin(), t.end(), toRemove[i]), t.end());
114 }
115 else
116 {
117 for(int i=0; i<toRemove.length(); i++)
118 std::replace(t.begin(), t.end(), toRemove[i], toReplace[i]);
119 }
120 }
121
122 // convert to a vector of words
123 std::vector<std::string> words;
124 std::stringstream ss(t);
125 std::string word;
126 while(ss >> word)
127 words.push_back(word);
128 return words;
129 }

318 static hoof::Tuple nanminmax(const hoof::vector2D<double>& vec)
319 {
320 double min = std::numeric_limits<double>::infinity();
321 double max = -std::numeric_limits<double>::infinity();
322 for(int i=0; i<vec.size(); i++)
323 {
324 for(int j=0; j<vec[i].size(); j++)
325 {
326 double value = vec[i][j];
327 if(!std::isnan(value))
328 {
329 if(value < min)
330 min = value;
331 if(value > max)
332 max = value;
333 }
334 }
335 }
336 if(std::isinf(min))
337 min = hoof::dNaN;
338 if(std::isinf(max))
339 max = hoof::dNaN;
340 hoof::Tuple result = {min, max};
341 return result;
